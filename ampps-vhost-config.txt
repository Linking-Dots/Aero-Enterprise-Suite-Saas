# AMPPS/Apache Virtual Host Configuration for aeos365.com
# Place this in your AMPPS Apache configuration or httpd.conf

<VirtualHost *:80>
    ServerName aeos365.com
    ServerAlias www.aeos365.com
    DocumentRoot "D:/Repos/Aero-HR/public"
    
    # Redirect HTTP to HTTPS
    RewriteEngine On
    RewriteCond %{HTTPS} off
    RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]
    
    <Directory "D:/Repos/Aero-HR/public">
        AllowOverride All
        Require all granted
        Options -Indexes
    </Directory>
    
    # Error and Access logs
    ErrorLog logs/aeos365_error.log
    CustomLog logs/aeos365_access.log combined
</VirtualHost>

<VirtualHost *:443>
    ServerName aeos365.com
    ServerAlias www.aeos365.com
    DocumentRoot "D:/Repos/Aero-HR/public"
    
    # SSL Configuration (adjust paths to your SSL certificates)
    SSLEngine on
    SSLCertificateFile "path/to/your/certificate.crt"
    SSLCertificateKeyFile "path/to/your/private.key"
    # SSLCertificateChainFile "path/to/your/ca-bundle.crt"
    
    <Directory "D:/Repos/Aero-HR/public">
        AllowOverride All
        Require all granted
        Options -Indexes
    </Directory>
    
    # Error and Access logs
    ErrorLog logs/aeos365_ssl_error.log
    CustomLog logs/aeos365_ssl_access.log combined
</VirtualHost>
